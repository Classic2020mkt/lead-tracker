callback+"("+out+")" : out ).setMimeType(ContentService.MimeType.JAVASCRIPT); } // ── doPost: reads from URLSearchParams (e.parameter) ───────────────────────── function doPost(e) { Logger.log("doPost HIT"); Logger.log("e.parameter: " + JSON.stringify(e.parameter)); var action = ""; var payload = {}; var raw = ""; try { // ── PRIMARY: URLSearchParams body (new frontend) ────────────────────── if (e.parameter && e.parameter.action) { action = e.parameter.action; raw = e.parameter.payload || "{}"; payload = JSON.parse(raw); Logger.log("Parsed via e.parameter — action: " + action); // ── FALLBACK: raw JSON body (old frontend / direct API calls) ───────── } else if (e.postData && e.postData.contents) { raw = e.postData.contents; payload = JSON.parse(raw); action = payload.loginLog ? Logger.log("saveAndSync done"); } } catch(err) { Logger.log("Handler ERROR: " + err.toString()); return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } // ── saveAndSync: merge + write lt_data row + sync readable tabs ─────────────── function saveAndSync(payload, raw) { var sheet = getSheet();
